8InlineDefine void star_midpoint(double* mid, const double* A, const double* B) {
9 double len;
10 double invlen;
11 // we don't divide by 2 because we immediately renormalize it...
12 mid[0] = A[0] + B[0];
13 mid[1] = A[1] + B[1];
14 mid[2] = A[2] + B[2];
15 //len = sqrt(square(mid[0]) + square(mid[1]) + square(mid[2]));
16 len = sqrt(mid[0] * mid[0] + mid[1] * mid[1] + mid[2] * mid[2]);
17 invlen = 1.0 / len;
18 mid[0] *= invlen;
19 mid[1] *= invlen;
20 mid[2] *= invlen;
21}

180WarnUnusedResult InlineDefine
181anbool star_coords(const double *s, const double *r,
182 anbool tangent, double *x, double *y) {
183 // As used by the sip.c code, this does the TAN projection
184 // (if "tangent" is TRUE; SIN projection otherwise)
185 // r: CRVAL
186 // s: RA,Dec to be projected
187 // ASSUME r,s are unit vectors
188 // sdotr: s dot r = |r||s| cos(theta) = cos(theta)
189 double sdotr = s[0] * r[0] + s[1] * r[1] + s[2] * r[2];
190 if (sdotr <= 0.0) {
191 // on the opposite side of the sky
192 return FALSE;
193 }
194 if (unlikely(r[2] == 1.0)) {
195 // North pole
196 double inv_s2 = 1.0 / s[2];
197 if (tangent) {
198 *x = s[0] * inv_s2;
199 *y = s[1] * inv_s2;
200 } else {
201 *x = s[0];
202 *y = s[1];
203 }
204 } else if (unlikely(r[2] == -1.0)) {
205 // South pole
206 double inv_s2 = 1.0 / s[2];
207 if (tangent) {
208 *x = s[0] * inv_s2;
209 *y = -s[1] * inv_s2;
210 } else {
211 *x = s[0];
212 *y = -s[1];
213 }
214 } else {
215 double etax, etay, xix, xiy, xiz, eta_norm;
216 double inv_en, inv_sdotr;
217 // eta is a vector perpendicular to r pointing in the direction
218 // of increasing RA. eta_z = 0 by definition.
219 etax = -r[1];
220 etay = r[0];
221 eta_norm = hypot(etax, etay);
222 inv_en = 1.0 / eta_norm;
223 etax *= inv_en;
224 etay *= inv_en;
225
226 // xi = r cross eta, a vector pointing northwards,
227 // in direction of increasing DEC
228 xix = -r[2] * etay;
229 xiy = r[2] * etax;
230 xiz = r[0] * etay - r[1] * etax;
231
232 // project s-r onto eta and xi. No need to subtract r from s, though,
233 // since eta and xi are orthogonal to r by construction.
234 *x = (s[0] * etax + s[1] * etay );
235 *y = (s[0] * xix + s[1] * xiy + s[2] * xiz);
236
237 // The "inv_sdotr" applies the TAN scaling
238 if (tangent) {
239 inv_sdotr = 1.0 / sdotr;
240 *x *= inv_sdotr;
241 *y *= inv_sdotr;
242 }
243 }
244 return TRUE;
245}

247InlineDefine Flatten Const double distsq2rad(double dist2) {
248 // cosine law: c^2 = a^2 + b^2 - 2 a b cos C
249 // c^2 is dist2. We want C.
250 // a = b = 1
251 // c^2 = 1 + 1 - 2 cos C
252 // dist2 = 2( 1 - cos C )
253 // 1 - (dist2 / 2) = cos C
254 // C = acos(1 - dist2 / 2)
255 return acos(1.0 - dist2 / 2.0);
256}
