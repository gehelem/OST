15// WCS TAN header.
16typedef struct {
17
18 // World coordinate of the tangent point, in ra,dec.
19 double crval[2];
20
21 // Tangent point location in pixel (CCD) coordinates
22 // This may not be in the image; consider a telescope with an array of
23 // CCD's, where the tangent point is in one or none of the CCD's.
24 double crpix[2];
25
26 // Matrix for the linear transformation of relative pixel coordinates
27 // (u,v) onto "intermediate world coordinates", which are in degrees
28 // (x,y). The x,y coordinates are on the tangent plane. If the SIP
29 // terms are all zero, then the equation to get from pixel coordinates
30 // to intermediate world coordinates is:
31 //
32 // u = pixel_x - crpix0
33 // v = pixel_y - crpix1
34 //
35 // x = [cd00 cd01] * u
36 // y [cd10 cd11] v
37 //
38 // where x,y are in intermediate world coordinates (i.e. x points
39 // along negative ra and y points to positive dec) and u,v are in pixel
40 // coordinates.
41 double cd[2][2];
42
43 // size of the image in pixels. Not strictly part of the WCS, but useful!
44 double imagew;
45 double imageh;
46
47 // SIN projection rather than TAN.
48 anbool sin;
49
50} tan_t;

52// Flat structure for minimal SIP wcs. This structure should contain enough
53// information to effectively represent any image, provided it is possible to
54// convert that image's projecton to a TAN projection and the distortion to SIP
55// distortion.
56typedef struct {
57
58 // A basic TAN header.
59 tan_t wcstan;
60
61 // Forward SIP coefficients
62 // The transformation from relative pixel coordinates to intermediate
63 // world coordinates[1] is:
64 //
65 // x = [cd00 cd01] * (u + f(u,v)) x,y are intermediate coordinates on the sky (in deg)
66 // y [cd10 cd11] (v + g(u,v)) which are just a "translation" away from final WCS
67 // u,v are original (unwarped) pixel coordinates
68 // where
69 // p q
70 // U = u + f(u,v) = u + SUM a[p][q] * u * v , p+q <= a_order
71 // p,q
72 //
73 // p q
74 // V = v + g(u,v) = v + SUM b[p][q] * u * v , p+q <= b_order
75 // p,q
76 //
77 // [1] The SIP convention for representing distortion in FITS image
78 // headers. D. L. Shupe, M.Moshir, J. Li, D. Makovoz, R. Narron, R. N.
79 // Hook. Astronomical Data Analysis Software and Systems XIV.
80 // http://ssc.spitzer.caltech.edu/postbcd/doc/shupeADASS.pdf
81 //
82 // Note: These matricies are larger than they strictly need to be
83 // because aij = 0 if i+j > a_order and similarily for b.
84 //
85 // Note: The convention for indicating that no SIP polynomial is
86 // present is to simply set [ab]_order to zero.
87 int a_order, b_order;
88 double a[SIP_MAXORDER][SIP_MAXORDER];
89 double b[SIP_MAXORDER][SIP_MAXORDER];
90
91 // Inverse SIP coefficients
92 // To convert from world coordinates back into image coordinates, the
93 // inverse transformation may be stored. To convert from intermediate
94 // world coordinates, first we calculate the linear pixel coordinates:
95 //
96 // -1
97 // U = [cd00 cd01] * x
98 // V [cd10 cd11] y
99 //
100 // Then, the original pixel coordinates are computed as:
101 //
102 // p q
103 // u = U + SUM ap[p][q] * U * V , p+q <= ap_order
104 // p,q
105 //
106 // p q
107 // v = V + SUM bp[p][q] * U * V , p+q <= ap_order
108 // p,q
109 //
110 // Note: ap_order does not necessarily equal a_order, because the
111 // inverse of a nth-order polynomial may be of higer order than n.
112 //
113 // Note: The convention for indicating that no inverse SIP polynomial
114 // is present is to simply set [ab]p_order to zero.
115 int ap_order, bp_order;
116 double ap[SIP_MAXORDER][SIP_MAXORDER];
117 double bp[SIP_MAXORDER][SIP_MAXORDER];
118} sip_t;
