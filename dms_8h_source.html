460inline void dms::SinCos(double &s, double &c) const
461{
462#ifdef PROFILE_SINCOS
463 std::clock_t start, stop;
464 start = std::clock();
465#endif
466
467#ifdef __GLIBC__
468#if (__GLIBC__ >= 2 && __GLIBC_MINOR__ >= 1 && !defined(__UCLIBC__))
469 //GNU version
470 sincos(radians(), &s, &c);
471#else
472 //For older GLIBC versions
473 s = ::sin(radians());
474 c = ::cos(radians());
475#endif
476#else
477 //ANSI-compliant version
478 s = ::sin(radians());
479 c = ::cos(radians());
480#endif
481
482#ifdef PROFILE_SINCOS
483 stop = std::clock();
484 seconds_in_trig += double(stop - start) / double(CLOCKS_PER_SEC);
485#endif
486
487#ifdef COUNT_DMS_SINCOS_CALLS
488 if (!m_sinCosCalled)
489 {
490 m_sinCosCalled = true;
491 ++dms_with_sincos_called;
492 }
493 if (m_sinDirty)
494 m_sinDirty = false;
495 else
496 ++redundant_trig_function_calls;
497
498 if (m_cosDirty)
499 m_cosDirty = false;
500 else
501 ++redundant_trig_function_calls;
502
503 trig_function_calls += 2;
504#endif
505}
