12/*
13 Computes the permutation array that will cause the "realarray" to be
14 sorted according to the "compare" function.
15
16 Ie, the first element in the sorted array will be at
17 (char*)realarray + perm[0] * array_stride
18
19 The "stride" parameter gives the number of bytes between successive entries
20 in "realarray".
21
22 If "perm" is NULL, a new permutation array will be allocated and returned.
23 Otherwise, the permutation array will be placed in "perm".
24
25 Note that if you pass in a non-NULL "perm" array, its existing values will
26 be used! You probably want to initialize it with "permutation_init()" to
27 set it to the identity permutation.
28 */
29int* permuted_sort(const void* realarray, int array_stride,
30 int (*compare)(const void*, const void*),
31 int* perm, int Nperm);
