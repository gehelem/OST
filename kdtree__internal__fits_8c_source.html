16static char* get_table_name(const char* treename, const char* tabname) {
17 char* rtn;
18 if (!treename) {
19 return strdup_safe(tabname);
20 }
21 asprintf_safe(&rtn, "%s_%s", tabname, treename);
22 return rtn;

25int MANGLE(kdtree_read_fits)(kdtree_fits_t* io, kdtree_t* kd) {
26 fitsbin_chunk_t chunk;
27
28 fitsbin_chunk_init(&chunk);
29
30 // kd->lr
31 chunk.tablename = get_table_name(kd->name, KD_STR_LR);
32 chunk.itemsize = sizeof(u32);
33 chunk.nrows = kd->nbottom;
34 chunk.required = FALSE;
35 if (kdtree_fits_read_chunk(io, &chunk) == 0) {
36 kd->lr = chunk.data;
37 }
38 free(chunk.tablename);
39
40 // kd->perm
41 chunk.tablename = get_table_name(kd->name, KD_STR_PERM);
42 chunk.itemsize = sizeof(u32);
43 chunk.nrows = kd->ndata;
44 chunk.required = FALSE;
45 if (kdtree_fits_read_chunk(io, &chunk) == 0) {
46 kd->perm = chunk.data;
47 }
48 free(chunk.tablename);
49
50 // kd->bb
51 chunk.tablename = get_table_name(kd->name, KD_STR_BB);
52 chunk.itemsize = sizeof(ttype) * kd->ndim * 2;
53 chunk.nrows = 0;
54 chunk.required = FALSE;
55 if (kdtree_fits_read_chunk(io, &chunk) == 0) {
56 int nbb_old = (kd->nnodes + 1) / 2 - 1;
57 int nbb_new = kd->nnodes;
58
59 // accept (but warn about) old-school buggy BB extension.
60 if (chunk.nrows == nbb_new) {
61 } else if (chunk.nrows == nbb_old) {
62 ERROR("Warning: this file contains an old, buggy, %s "
63 "extension; it has %i rather than %i items. Proceeding "
64 "anyway, but this is probably going to cause problems!",
65 chunk.tablename, nbb_old, nbb_new);
66 } else {
67 // error?
68 ERROR("Bounding-box table %s should contain either %i (new) or "
69 "%i (old) bounding-boxes, but it has %i.",
70 chunk.tablename, nbb_new, nbb_old, chunk.nrows);
71 free(chunk.tablename);
72 return -1;
73 }
74 kd->bb.any = chunk.data;
75 kd->n_bb = chunk.nrows;
76 }
77 free(chunk.tablename);
78
79 // kd->split
80 chunk.tablename = get_table_name(kd->name, KD_STR_SPLIT);
81 chunk.itemsize = sizeof(ttype);
82 chunk.nrows = kd->ninterior;
83 chunk.required = FALSE;
84 if (kdtree_fits_read_chunk(io, &chunk) == 0) {
85 kd->split.any = chunk.data;
86 }
87 free(chunk.tablename);
88
89 // kd->splitdim
90 chunk.tablename = get_table_name(kd->name, KD_STR_SPLITDIM);
91 chunk.itemsize = sizeof(u8);
92 chunk.nrows = kd->ninterior;
93 chunk.required = FALSE;
94 if (kdtree_fits_read_chunk(io, &chunk) == 0) {
95 kd->splitdim = chunk.data;
96 }
97 free(chunk.tablename);
98
99 // kd->data
100 chunk.tablename = get_table_name(kd->name, KD_STR_DATA);
101 chunk.itemsize = sizeof(dtype) * kd->ndim;
102 chunk.nrows = kd->ndata;
103 chunk.required = TRUE;
104 if (kdtree_fits_read_chunk(io, &chunk) == 0) {
105 kd->data.any = chunk.data;
106 }
107 free(chunk.tablename);
108
109 // kd->minval/kd->maxval/kd->scale
110 chunk.tablename = get_table_name(kd->name, KD_STR_RANGE);
111 chunk.itemsize = sizeof(double);
112 chunk.nrows = (kd->ndim * 2 + 1);
113 chunk.required = FALSE;
114 if (kdtree_fits_read_chunk(io, &chunk) == 0) {
115 double* r;
116 r = chunk.data;
117 kd->minval = r;
118 kd->maxval = r + kd->ndim;
119 kd->scale = r[kd->ndim * 2];
120 kd->invscale = 1.0 / kd->scale;
121 }
122 free(chunk.tablename);
123
124 if (!(kd->bb.any ||
125 (kd->split.any && (TTYPE_INTEGER || kd->splitdim)))) {
126 ERROR("kdtree contains neither bounding boxes nor split+dim data");
127 return -1;
128 }
129
130 if ((TTYPE_INTEGER && !ETYPE_INTEGER) &&
131 !(kd->minval && kd->maxval)) {
132 ERROR("treee does not contain required range information");
133 return -1;
134 }
135
136 if (kd->split.any) {
137 if (kd->splitdim)
138 kd->splitmask = UINT32_MAX;
139 else
140 compute_splitbits(kd);
141 }
142
143 return 0;
144}
