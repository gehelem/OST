168struct kdtree {
169 /*
170 A bitfield describing the type of this tree.
171 */
172 u32 treetype;
173
174 int32_t* lr; /* Points owned by leaf nodes, stored and manipulated
175 in a way that's too complicated to explain in this comment.
176 (nbottom) */
177
178 u32* perm; /* Permutation index / hairstyle from the 80s
179 (ndata) */
180
181 /* Bounding box: list of D-dimensional lower hyperrectangle corner followed by
182 D-dimensional upper corner.
183 (ttype x ndim x nnodes) */
184 union {
185 float* f;
186 double* d;
187 u32* u;
188 u16* s;
189 void* any;
190 } bb;
191 // how many bounding-boxes were found?
192 int n_bb;
193
194 /* Split position (& dimension for ints) (ttype x ninterior). */
195 union {
196 float* f;
197 double* d;
198 u32* u;
199 u16* s;
200 void* any;
201 } split;
202
203 /* Split dimension for floating-point types (x ninterior) */
204 u8* splitdim;
205
206 /* bitmasks for the split dimension and location. */
207 u8 dimbits;
208 u32 dimmask;
209 u32 splitmask;
210
211 /* Raw coordinate data as xyzxyzxyz (dtype) */
212 union {
213 float* f;
214 double* d;
215 u32* u;
216 u16* s;
217 void* any;
218 } data;
219
220 /* Should the kdtree free(kd->data) when kdtree_free(kd) is called?
221
222 Normally the kdtree does *not* free the data, unless the datatype
223 had to be converted (if "external" type != "data" type) so that a
224 new array had to be allocated.
225 */
226 int free_data;
227
228 double* minval;
229 double* maxval;
230 double scale; /* kdtype per real -- isotropic */
231 double invscale; /* real per kdtype */
232
233 int ndata; /* Number of items */
234 int ndim; /* Number of dimensions */
235 int nnodes; /* Number of nodes */
236 int nbottom; /* Number of leaf nodes */
237 int ninterior; /* Number of internal nodes */
238 int nlevels;
239
240 int has_linear_lr;
241
242 // For i/o: the name of this tree in the file.
243 char* name;
244
245 void* io;
246
247 struct kdtree_funcs fun;
248};
