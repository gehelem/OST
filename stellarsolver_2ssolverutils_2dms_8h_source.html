447inline void dms::SinCos(double &s, double &c) const
448{
449#ifdef PROFILE_SINCOS
450 std::clock_t start, stop;
451 start = std::clock();
452#endif
453
454#ifdef __GLIBC__
455#if (__GLIBC__ >= 2 && __GLIBC_MINOR__ >= 1 && !defined(__UCLIBC__))
456 //GNU version
457 sincos(radians(), &s, &c);
458#else
459 //For older GLIBC versions
460 s = ::sin(radians());
461 c = ::cos(radians());
462#endif
463#else
464 //ANSI-compliant version
465 s = ::sin(radians());
466 c = ::cos(radians());
467#endif
468
469#ifdef PROFILE_SINCOS
470 stop = std::clock();
471 seconds_in_trig += double(stop - start) / double(CLOCKS_PER_SEC);
472#endif
473
474#ifdef COUNT_DMS_SINCOS_CALLS
475 if (!m_sinCosCalled)
476 {
477 m_sinCosCalled = true;
478 ++dms_with_sincos_called;
479 }
480 if (m_sinDirty)
481 m_sinDirty = false;
482 else
483 ++redundant_trig_function_calls;
484
485 if (m_cosDirty)
486 m_cosDirty = false;
487 else
488 ++redundant_trig_function_calls;
489
490 trig_function_calls += 2;
491#endif
492}
