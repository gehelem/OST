725// Helper function to increment all placeholder numbers in message
726// Example: "value %1 max %2" becomes "value %2 max %3"
727inline QString incrementPlaceholders(const QString &msg)
728{
729 QString result = msg;
730 // Replace from highest to lowest to avoid conflicts (e.g., %1→%2 shouldn't affect original %2)
731 for (int i = 99; i >= 1; --i)
732 {
733 QString oldPlaceholder = QString("%%1").arg(i);
734 QString newPlaceholder = QString("%%1").arg(i + 1);
735 result = result.replace(oldPlaceholder, newPlaceholder);
736 }
737 return result;
738}
