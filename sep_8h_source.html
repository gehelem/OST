213/* Sum array values within a circular aperture.
214 *
215 * Notes
216 * -----
217 * error : Can be a scalar (default), an array, or NULL
218 * If an array, set the flag SEP_ERROR_IS_ARRAY in `inflags`.
219 * Can represent 1-sigma std. deviation (default) or variance.
220 * If variance, set the flag SEP_ERROR_IS_VARIANCE in `inflags`.
221 *
222 * gain : If 0.0, poisson noise on sum is ignored when calculating error.
223 * Otherwise, (sum / gain) is added to the variance on sum.

298/* sep_kron_radius()
299 *
300 * Calculate Kron radius within an ellipse given by
301 *
302 * cxx*(x'-x)^2 + cyy*(y'-y)^2 + cxy*(x'-x)*(y'-y) < r^2
303 *
304 * The Kron radius is sum(r_i * v_i) / sum(v_i) where v_i is the value of pixel
305 * i and r_i is the "radius" of pixel i, as given by the left hand side of
306 * the above equation.
307 *
308 * Flags that might be set:
309 * SEP_APER_HASMASKED - at least one of the pixels in the ellipse is masked.
310 * SEP_APER_ALLMASKED - All pixels in the ellipse are masked. kronrad = 0.
311 * SEP_APER_NONPOSITIVE - There was a nonpositive numerator or deminator.
312 * kronrad = 0.
313 */
314int sep_kron_radius(sep_image *im, double x, double y,
315 double cxx, double cyy, double cxy, double r, int id,
316 double *kronrad, short *flag);
